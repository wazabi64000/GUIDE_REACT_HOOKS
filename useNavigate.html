useNavigate — explication simple

Idée : useNavigate permet de changer de page par code (par ex. après un login). Ici, on montre la logique : effectuer une action puis rediriger automatiquement.

Exemple React (snippet)

// ExempleUseNavigate.jsx
import React, { useState } from "react";
import { useNavigate } from "react-router-dom";

export default function ExempleUseNavigate() {
  const [name, setName] = useState("");
  const navigate = useNavigate();

  function handleLogin(e) {
    e.preventDefault();
    if (!name.trim()) return;
    // Ici on ferait login(name)
    navigate("/profile"); // redirection par code
  }

  return (
    <form onSubmit={handleLogin}>
      <input value={name} onChange={(e) => setName(e.target.value)} />
      <button type="submit">Login & Go to profile</button>
    </form>
  );
}

Logique : `navigate('/profile')` provoque une navigation programmée — très utile après une action (login, envoi formulaire, suppression d'un élément...).
